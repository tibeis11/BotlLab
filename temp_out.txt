-- Migration Script for Ingredients Engine v2

-- 1. Create Tables
CREATE TABLE IF NOT EXISTS ingredient_master (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name          TEXT NOT NULL,
  type          TEXT NOT NULL CHECK (type IN ('malt','hop','yeast','misc','water')),
  aliases       TEXT[] DEFAULT '{}',
  description   TEXT,
  created_at    TIMESTAMPTZ DEFAULT now(),
  updated_at    TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_ingredient_master_type ON ingredient_master(type);
CREATE INDEX IF NOT EXISTS idx_ingredient_master_aliases ON ingredient_master USING GIN(aliases);

CREATE TABLE IF NOT EXISTS ingredient_products (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  master_id     UUID NOT NULL REFERENCES ingredient_master(id) ON DELETE RESTRICT,
  name          TEXT NOT NULL,
  manufacturer  TEXT,
  color_ebc     NUMERIC(7,2),
  potential_pts NUMERIC(6,2),
  alpha_pct     NUMERIC(5,2),
  beta_pct      NUMERIC(5,2),
  cohumulone_pct NUMERIC(5,2),
  attenuation_pct NUMERIC(5,2),
  flocculation    TEXT CHECK (flocculation IN ('low','medium','high','very_high')),
  min_temp_c      NUMERIC(4,1),
  max_temp_c      NUMERIC(4,1),
  alcohol_tolerance_pct NUMERIC(4,1),
  notes           TEXT,
  source_url      TEXT,
  is_verified     BOOLEAN DEFAULT false,
  created_at      TIMESTAMPTZ DEFAULT now(),
  updated_at      TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_ingredient_products_master ON ingredient_products(master_id);

CREATE TABLE IF NOT EXISTS ingredient_import_queue (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  raw_name        TEXT NOT NULL,
  type            TEXT,
  raw_data        JSONB,
  suggested_master_id UUID REFERENCES ingredient_master(id),
  imported_by     UUID REFERENCES auth.users(id),
  status          TEXT DEFAULT 'pending' CHECK (status IN ('pending','merged','rejected')),
  created_at      TIMESTAMPTZ DEFAULT now()
);

CREATE TABLE IF NOT EXISTS recipe_ingredients (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  recipe_id       UUID NOT NULL REFERENCES brews(id) ON DELETE CASCADE,
  product_id      UUID REFERENCES ingredient_products(id),
  master_id       UUID REFERENCES ingredient_master(id),
  raw_name        TEXT,
  type            TEXT,
  amount          NUMERIC(10,3),
  unit            TEXT,
  time_minutes    INTEGER,
  usage           TEXT,
  override_alpha  NUMERIC(5,2),
  override_color_ebc NUMERIC(7,2),
  override_attenuation NUMERIC(5,2),
  sort_order      INTEGER DEFAULT 0,
  created_at      TIMESTAMPTZ DEFAULT now()
);

CREATE INDEX IF NOT EXISTS idx_recipe_ingredients_recipe ON recipe_ingredients(recipe_id);

-- 2. Seed Empty Masters (Fallback for Unmatched)
-- Using fixed UUIDs for deterministic fallback mapping
INSERT INTO ingredient_master (id, name, type, aliases, description)
VALUES

  ('00000000-0000-4000-a000-000000000002', 'Unbekannter Hopfen', 'hop', ARRAY['Unknown Hop', 'Sonstiges Hopfen'], 'Fallback für nicht zuerkennbare Hopfen-Imports'),
  ('00000000-0000-4000-a000-000000000003', 'Unbekannte Hefe', 'yeast', ARRAY['Unknown Yeast', 'Sonstiges Hefe'], 'Fallback für nicht zuerkennbare Hefe-Imports'),
  ('00000000-0000-4000-a000-000000000004', 'Unbekannte Zutat', 'misc', ARRAY['Unknown Misc', 'Sonstige Zutat'], 'Fallback für nicht zuerkennbare Misc-Imports')

-- 3. Data Extraction (JSONB -> Relational)
DO $$ 
DECLARE
  brew_row RECORD;
  malt_item JSONB;
  hop_item JSONB;
  yeast_item JSONB;
BEGIN
  -- Nur Brews verarbeiten die noch JSONB-Zutaten haben (idempotent)
  FOR brew_row IN SELECT id, data FROM brews WHERE data IS NOT NULL AND (data ? 'malts' OR data ? 'hops' OR data ? 'yeast') LOOP
    
    -- Extract Malts
    IF jsonb_typeof(brew_row.data->'malts') = 'array' THEN
      FOR malt_item IN SELECT * FROM jsonb_array_elements(brew_row.data->'malts') LOOP
        INSERT INTO recipe_ingredients (recipe_id, master_id, raw_name, type, amount, unit, override_color_ebc)
        VALUES (
          brew_row.id,
          '00000000-0000-4000-a000-000000000001', -- map directly to Unknown Malt for v1 migration
          malt_item->>'name',
          'malt',
          (REPLACE(NULLIF(malt_item->>'amount', ''), ',', '.))::NUMERIC,
          COALESCE(malt_item->>'unit', 'kg'),
          (REPLACE(NULLIF(malt_item->>'color', ''), ',', '.))::NUMERIC
        );
      END LOOP;
    END IF;

    -- Extract Hops
    IF jsonb_typeof(brew_row.data->'hops') = 'array' THEN
      FOR hop_item IN SELECT * FROM jsonb_array_elements(brew_row.data->'hops') LOOP
        INSERT INTO recipe_ingredients (recipe_id, master_id, raw_name, type, amount, unit, time_minutes, usage, override_alpha)
        VALUES (
          brew_row.id,
          '00000000-0000-4000-a000-000000000002', -- map directly to Unknown Hop
          hop_item->>'name',
          'hop',
          (REPLACE(NULLIF(hop_item->>'amount', ''), ',', '.))::NUMERIC,
          COALESCE(hop_item->>'unit', 'g'),
          (NULLIF(hop_item->>'time', ''))::INTEGER,
          hop_item->>'usage',
          (REPLACE(NULLIF(hop_item->>'alpha', ''), ',', '.))::NUMERIC
        );
      END LOOP;
    END IF;

    -- Extract Yeast
    IF jsonb_typeof(brew_row.data->'yeast') = 'array' THEN
      FOR yeast_item IN SELECT * FROM jsonb_array_elements(brew_row.data->'yeast') LOOP
        INSERT INTO recipe_ingredients (recipe_id, master_id, raw_name, type, amount, unit, override_attenuation)
        VALUES (
          brew_row.id,
          '00000000-0000-4000-a000-000000000003', -- map directly to Unknown Yeast
          yeast_item->>'name',
          'yeast',
          (REPLACE(NULLIF(yeast_item->>'amount', ''), ',', '.))::NUMERIC,
          COALESCE(yeast_item->>'unit', 'pkg'),
          (REPLACE(NULLIF(yeast_item->>'attenuation', ''), ',', '.))::NUMERIC
        );
      END LOOP;
    END IF;

  END LOOP;
END 
$$;

-- 4. Delete the Extracted Keys from JSONB
UPDATE brews
SET data = data - 'malts' - 'hops' - 'yeast'
WHERE data ? 'malts' OR data ? 'hops' OR data ? 'yeast';

-- 5. RLS Policies
ALTER TABLE ingredient_master ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read" ON ingredient_master FOR SELECT USING (true);

ALTER TABLE ingredient_products ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public read" ON ingredient_products FOR SELECT USING (true);

ALTER TABLE ingredient_import_queue ENABLE ROW LEVEL SECURITY;
CREATE POLICY "own entries" ON ingredient_import_queue FOR SELECT USING (
  imported_by = auth.uid() OR
  (auth.jwt() ->> 'role') = 'service_role'
);
CREATE POLICY "authenticated insert" ON ingredient_import_queue FOR INSERT WITH CHECK (
  auth.uid() IS NOT NULL
);

ALTER TABLE recipe_ingredients ENABLE ROW LEVEL SECURITY;
CREATE POLICY "public recipe read" ON recipe_ingredients FOR SELECT USING (
  EXISTS (SELECT 1 FROM brews WHERE id = recipe_ingredients.recipe_id AND is_public = true)
  OR EXISTS (SELECT 1 FROM brews WHERE id = recipe_ingredients.recipe_id AND user_id = auth.uid())
);
CREATE POLICY "owner write" ON recipe_ingredients FOR ALL USING (
  EXISTS (SELECT 1 FROM brews WHERE id = recipe_ingredients.recipe_id AND user_id = auth.uid())
);
